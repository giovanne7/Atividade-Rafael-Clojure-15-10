(ns sistema-notas.core)

;; Limite de aprovação
(def limite-aprovacao 60)

;; Mensagens de desempenho
(def mensagens
  {:excelente "Turma excelente!"
   :bom       "Bom desempenho!"
   :ruim      "É necessário melhorar!"})

;; Função para definir o conceito de cada nota
(defn conceito [nota]
  (cond
    (>= nota 90) "A"
    (>= nota 80) "B"
    (>= nota 70) "C"
    (>= nota 60) "D"
    :else        "F"))

;; Lista fixa de alunos (nome e nota)
(def alunos [["Ana" 95]
             ["Bruno" 72]
             ["Carla" 55]])

;; Função principal
(defn -main []
  ;; Exibir conceito de cada aluno
  (doseq [[nome nota] alunos]
    (println (str nome " - Conceito: " (conceito nota))))

  ;; Calcular média
  (let [notas (map second alunos)
        soma (reduce + notas)
        qtd (count alunos)
        media (/ soma qtd)
        ;; Contar aprovados
        aprovados (count (filter #(>= % limite-aprovacao) notas))
        ;; Mensagem de desempenho
        desempenho (cond
                     (>= media 80) (:excelente mensagens)
                     (>= media 60) (:bom mensagens)
                     :else (:ruim mensagens))]
    ;; Exibir estatísticas da turma
    (println "\nMédia da turma:" media)
    (println "Aprovados:" aprovados)
    (println "Desempenho geral:" desempenho)))

;; Executar
(-main)
